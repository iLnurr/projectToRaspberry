#How to get file from google drive to tomcat folder in raspbian
wget --no-check-certificate 'https://drive.google.com/uc?export=download&id=0B6iC2KVuzaXzNFc5dTBYYm5NOW8' -O ROOT.war

sudo su

cd /usr/share/apache-tomcat-8.5.5/bin
./shutdown.sh

cd ..

cd webapps

rm -r ROOT
rm ROOT.war

cp -i /home/pi/ROOT.war /usr/share/apache-tomcat-8.5.5/webapps

cd ..
cd bin
./startup.sh
